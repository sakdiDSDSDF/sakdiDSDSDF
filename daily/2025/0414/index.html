    二分¶

    https://www.luogu.com.cn/problem/P2678¶

    读题要求最大最小值，触发关键词直接二分答案，从零二分到最大距离，然后去遍历数组中各个石头之间的距离判断是否小于二分值即可，如果小于则需要移除石头并记录操作次数加一，最后根据操作次数与题目所给最大数对比判断是否check函数返回值。

    /*
   (｡･ω･｡)ﾉ♡  我喜欢斯卡蒂啊，我喜欢流萤啊
*/
#include<bits/stdc++.h>
#define INF 1e18
#define pb push_back
#define ll long long
using namespace std;
bool check(vector<ll> ANS,ll MID,int M){
    int cnt=0;
    for(int i=0;i<ANS.size()-1;i++){
        ll juli=ANS[i+1]-ANS[i];
        if(juli<MID){
            cnt++;
            ANS[i+1]=ANS[i];
        }
        if(cnt>=M+1) break;
    }
    // for(int i=0;i<ANS.size();i++){
    //     cout<<ANS[i]<<" ";
    // }
    // cout<<endl;
    return cnt>M;
}
int main(){
    int n,m;
    ll l;
    cin>>l;
    cin>>n>>m;
    if(!n){
        cout<<l<<endl;
        return 0;
    }
    vector<ll> dis(n+2);
    for(int i=1;i<=n;i++){
        cin>>dis[i];
    }
    dis[0]=0,dis[n+1]=l;
    ll low=1,high=l;
    while(low<high){
        ll mid=(low+high)>>1;
        if(check(dis,mid,m)){
            high=mid;
        }
        else{
            low=mid+1;
        }
        //cout<<low<<" "<<high<<endl;
    }
    cout<<low-1<<endl;
    return 0;
}

    https://www.luogu.com.cn/problem/P1902¶

    看题目同样要求的是一个最小的最大值，同样直接二分答案，从零二分到最大的合法答案，然后用dfs走一遍这个二维数组，比二分值大的就不走，只走小于等于二分值的，最后判断一下dfs能否到达最后一层即可

    /*
   (｡･ω･｡)ﾉ♡  我喜欢斯卡蒂啊，我喜欢流萤啊
*/
#include<bits/stdc++.h>
#define INF 1e18
#define pb push_back
#define ll long long
using namespace std;
bool ok=false;
int n,m;
vector<vector<bool>> vis(n,vector<bool> (m,false));
void dfs(int mid,vector<vector<int>> &p,int x,int y){
    if(x==n-1){
        ok=true;
        return;
    }
    if(x==0&&y==0){
        if(x==n-1){
            ok=true;
            return;
        }
        if(!vis[x+1][y]&&p[x+1][y]<=mid){
            vis[x+1][y]=true;
            dfs(mid,p,x+1,y);
        }
        if(!vis[x][y+1]&&p[x][y+1]<=mid){
            vis[x][y+1]=true;
            dfs(mid,p,x,y+1);
        }
    }
    else if(x==0&&y==m-1){
        if(x==n-1){
            ok=true;
            return;
        }
        if(!vis[x+1][y]&&p[x+1][y]<=mid){
            vis[x+1][y]=true;
            dfs(mid,p,x+1,y);
        }
        if(!vis[x][y-1]&&p[x][y-1]<=mid){
            vis[x][y-1]=true;
            dfs(mid,p,x,y-1);
        }
    }
    else if(x==0){
        if(x==n-1){
            ok=true;
            return;
        }
        if(!vis[x+1][y]&&p[x+1][y]<=mid){
            vis[x+1][y]=true;
            dfs(mid,p,x+1,y);
        }
        if(!vis[x][y+1]&&p[x][y+1]<=mid){
            vis[x][y+1]=true;
            dfs(mid,p,x,y+1);
        }
        if(!vis[x][y-1]&&p[x][y-1]<=mid){
            vis[x][y-1]=true;
            dfs(mid,p,x,y-1);
        }
    }
    else if(y==0){
        if(x==n-1){
            ok=true;
            return;
        }
        if(!vis[x+1][y]&&p[x+1][y]<=mid){
            vis[x+1][y]=true;
            dfs(mid,p,x+1,y);
        }
        if(!vis[x-1][y]&&p[x-1][y]<=mid){
            vis[x-1][y]=true;
            dfs(mid,p,x-1,y);
        }
        if(!vis[x][y+1]&&p[x][y+1]<=mid){
            vis[x][y+1]=true;
            dfs(mid,p,x,y+1);
        }
    }
    else if(y==m-1){
        if(x==n-1){
            ok=true;
            return;
        }
        if(!vis[x+1][y]&&p[x+1][y]<=mid){
            vis[x+1][y]=true;
            dfs(mid,p,x+1,y);
        }
        if(!vis[x-1][y]&&p[x-1][y]<=mid){
            vis[x-1][y]=true;
            dfs(mid,p,x-1,y);
        }
        if(!vis[x][y-1]&&p[x][y-1]<=mid){
            vis[x][y-1]=true;
            dfs(mid,p,x,y-1);
        }
    }
    else{
        if(x==n-1){
            ok=true;
            return;
        }
        if(!vis[x+1][y]&&p[x+1][y]<=mid){
            vis[x+1][y]=true;
            dfs(mid,p,x+1,y);
        }
        if(!vis[x-1][y]&&p[x-1][y]<=mid){
            vis[x-1][y]=true;
            dfs(mid,p,x-1,y);
        }
        if(!vis[x][y-1]&&p[x][y-1]<=mid){
            vis[x][y-1]=true;
            dfs(mid,p,x,y-1);
        }
        if(!vis[x][y+1]&&p[x][y+1]<=mid){
            vis[x][y+1]=true;
            dfs(mid,p,x,y+1);
        }
    }
}
bool check(int mid,vector<vector<int>> &p){
    int x=0,y=0;
    vis=vector<vector<bool>> (n,vector<bool> (m));
    ok=false;
    dfs(mid,p,x,y);
    return ok;
}
int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>n>>m;  
    vector<vector<int>> p(n,vector<int> (m));
    int l=0,r=0;
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            cin>>p[i][j];
            r=max(p[i][j],r);
        }
    }
    while(l<r){
        int mid=(l+r)>>1;
        if(check(mid,p)){
            r=mid;
        }
        else{
            l=mid+1;
        }
    } 
    cout<<l<<endl;
    return 0;
}
